# Create a pet

<Endpoint method="POST" path={"/pets"} />

<ParameterTable
  title="Body"
  parameters={[
    {
      description: "The pet's name.",
      name: "name",
      required: true,
      type: "string",
    },
    {
      description: "An optional tag.",
      name: "tag",
      required: false,
      type: "string",
    },
  ]}
/>

<RequestExample>

```json
{
  "name": "Rex",
  "tag": "dog"
}
```

</RequestExample>

<ResponseExample title="201 — The created pet.">

```json
{
  "id": 1,
  "name": "Rex"
}
```

</ResponseExample>
